% job listing environments
\newcommand*{\exptitle}[2]{
    \begin{tabularx}{\linewidth}{@{}l X r@{}}
        \textbf{#1} & \hfill &  #2 \\[5pt]
    \end{tabularx}
}

\newcommand*{\exptitlelong}[4]{
    \begin{tabularx}{\linewidth}{@{}l X r@{}}
        \textbf{#1} & \hfill & #2 \\
        \textbf{#3} & \hfill & \textit{#4} \\[5pt]
    \end{tabularx}
}

% block that is left-aligned and only reaches to a certain width
\newenvironment{expblock}[1][0.75\linewidth]
{
    \begin{minipage}{#1}
}
{
    \end{minipage}
    \expspace
}

\newenvironment{explist}
{
    \begin{itemize}[nosep,after=\strut, leftmargin=1em, itemsep=3pt]%, label=--]
}
{
    \end{itemize}
}

% single-row header entry with optional body content in a minipage
% usage: \exptitlecompact[body][minipage-width][degree-col-width]{title}{institution}{dates}
% All three optional args have defaults; body is free-form.
% degree-col-width is a fixed-width column so institution names align across entries.
\NewDocumentCommand{\exptitlecompact}{O{} O{0.75\linewidth} O{0.44\linewidth} m m m}{%
    \begin{tabularx}{\linewidth}{@{}p{#3} @{\hspace{1.5em}} l X r@{}}
        \textbf{#4} & \textbf{#5} & & #6 \\[2pt]
    \end{tabularx}\par\vspace{-\parskip}%
    \IfBlankTF{#1}{}{%
        \begin{minipage}[t]{#2}%
            #1%
        \end{minipage}%
    }%
    \expspace%
}

\newcommand*{\expspace}{\vspace{5pt}}
